<template>
  <div>
    <el-row :gutter="20">
      <el-col :span="12" :offset="6">
        <div class="grid-content bg-purple">
          Đăng nhập
          <el-form
            :model="ruleForm"
            status-icon
            :rules="rules"
            ref="datalogin"
            label-width="120px"

              <el-input v-model="ruleForm.email"></el-input>

            message: "Hãy nhập Email",

          }

        password: [
          {
            required: true,
            message: "Hãy nhập mật khẩu",

      },

  methods: {

      this.$refs['datalogin'].validate((valid) => {
        if (valid) {
          alert("submit!");
        } else {
          return false;
        }
      });
    },

  },
};
</script>
